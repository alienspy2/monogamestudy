48// USAGE
49//
50// Include this file in whatever places neeed to refer to it. In ONE C/C++
51// file, write:
52// #define STB_TRUETYPE_IMPLEMENTATION
53// before the #include of this file. This expands out the actual
54// implementation into that C/C++ file.
55//
56// Look at the header-file sections below for the API, but here's a quick skim:
57//
58// Simple 3D API (don't ship this, but it's fine for tools and quick start,
59// and you can cut and paste from it to move to more advanced)
60// stbtt_BakeFontBitmap() -- bake a font to a bitmap for use as texture
61// stbtt_GetBakedQuad() -- compute quad to draw for a given char
62//
63// "Load" a font file from a memory buffer (you have to keep the buffer loaded)
64// stbtt_InitFont()
65// stbtt_GetFontOffsetForIndex() -- use for TTC font collections
66//
67// Render a unicode codepoint to a bitmap
68// stbtt_GetCodepointBitmap() -- allocates and returns a bitmap
69// stbtt_MakeCodepointBitmap() -- renders into bitmap you provide
70// stbtt_GetCodepointBitmapBox() -- how big the bitmap must be
71//
72// Character advance/positioning
73// stbtt_GetCodepointHMetrics()
74// stbtt_GetFontVMetrics()
75// stbtt_GetCodepointKernAdvance()
76//
77// ADVANCED USAGE
78//
79// Quality:
80//
81// - Use the functions with Subpixel at the end to allow your characters
82// to have subpixel positioning. Since the font is anti-aliased, not
83// hinted, this is very import for quality.
84//
85// - Kerning is now supported, and if you're supporting subpixel rendering
86// then kerning is worth using to give your text a polished look.
87//
88// Performance:
89//
90// - Convert Unicode codepoints to "glyphs" and operate on the glyphs; if
91// you don't do this, stb_truetype is forced to do the conversion on
92// every call.
93//
94// - There are a lot of memory allocations. We should modify it to take
95// a temp buffer and allocate from the temp buffer (without freeing),
96// should help performance a lot.
97//
98// NOTES
99//
100// The system uses the raw data found in the .ttf file without changing it
101// and without building auxiliary data structures. This is a bit inefficient
102// on little-endian systems (the data is big-endian), but assuming you're
103// caching the bitmaps or glyph shapes this shouldn't be a big deal.
104//
105// It appears to be very hard to programmatically determine what font a
106// given file is in a general way. I provide an API for this, but I don't
107// recommend it.
108//

557//
558// Finding the right font...
559//
560// You should really just solve this offline, keep your own tables
561// of what font is what, and don't try to get it out of the .ttf file.
562// That's because getting it out of the .ttf file is really hard, because
563// the names in the file can appear in many possible encodings, in many
564// possible languages, and e.g. if you need a case-insensitive comparison,
565// the details of that depend on the encoding & language in a complex way
566// (actually underspecified in truetype, but also gigantic).
567//
568// But you can use the provided functions in two possible ways:

696static int stbtt__isfont(const stbtt_uint8 *font)

706// @OPTIMIZE: binary search
707static stbtt_uint32 stbtt__find_table(stbtt_uint8 *data, stbtt_uint32 fontstart, const char *tag)
708{
709 stbtt_int32 num_tables = ttUSHORT(data+fontstart+4);
710 stbtt_uint32 tabledir = fontstart + 12;
711 stbtt_int32 i;
712 for (i=0; i < num_tables; ++i) {
713 stbtt_uint32 loc = tabledir + 16*i;
714 if (stbtt_tag(data+loc+0, tag))
715 return ttULONG(data+loc+8);
716 }
717 return 0;
718}

720int stbtt_GetFontOffsetForIndex(const unsigned char *font_collection, int index)
721{
722 // if it's just a font, there's only one valid index
723 if (stbtt__isfont(font_collection))
724 return index == 0 ? 0 : -1;
725
726 // check if it's a TTC
727 if (stbtt_tag(font_collection, "ttcf")) {
728 // version 1?

730 stbtt_int32 n = ttLONG(font_collection+8);

732 return -1;
733 return ttULONG(font_collection+12+index*14);
734 }
735 }
736 return -1;

739int stbtt_InitFont(stbtt_fontinfo *info, const unsigned char *data2, int fontstart)
740{
741 stbtt_uint8 *data = (stbtt_uint8 *) data2;
742 stbtt_uint32 cmap, t;
743 stbtt_int32 i,numTables;
744
745 info->data = data;
746 info->fontstart = fontstart;
747
748 cmap = stbtt__find_table(data, fontstart, "cmap"); // required
749 info->loca = stbtt__find_table(data, fontstart, "loca"); // required
750 info->head = stbtt__find_table(data, fontstart, "head"); // required
751 info->glyf = stbtt__find_table(data, fontstart, "glyf"); // required
752 info->hhea = stbtt__find_table(data, fontstart, "hhea"); // required
753 info->hmtx = stbtt__find_table(data, fontstart, "hmtx"); // required
754 info->kern = stbtt__find_table(data, fontstart, "kern"); // not required
755 if (!cmap || !info->loca || !info->head || !info->glyf || !info->hhea || !info->hmtx)
756 return 0;
757
758 t = stbtt__find_table(data, fontstart, "maxp");
759 if (t)
760 info->numGlyphs = ttUSHORT(data+t+4);
761 else
762 info->numGlyphs = 0xffff;
763
764 // find a cmap encoding table we understand *now* to avoid searching
765 // later. (todo: could make this installable)
766 // the same regardless of glyph.
767 numTables = ttUSHORT(data + cmap + 2);
768 info->index_map = 0;
769 for (i=0; i < numTables; ++i) {
770 stbtt_uint32 encoding_record = cmap + 4 + 8 * i;
771 // find an encoding we understand:
772 switch(ttUSHORT(data+encoding_record)) {
773 case STBTT_PLATFORM_ID_MICROSOFT:
774 switch (ttUSHORT(data+encoding_record+2)) {
775 case STBTT_MS_EID_UNICODE_BMP:
776 case STBTT_MS_EID_UNICODE_FULL:
777 // MS/Unicode
778 info->index_map = cmap + ttULONG(data+encoding_record+4);
779 break;
780 }
781 break;
782 }
783 }
784 if (info->index_map == 0)
785 return 0;
786
787 info->indexToLocFormat = ttUSHORT(data+info->head + 50);
788 return 1;
789}

791int stbtt_FindGlyphIndex(const stbtt_fontinfo *info, int unicode_codepoint)
792{
793 stbtt_uint8 *data = info->data;
794 stbtt_uint32 index_map = info->index_map;
795
796 stbtt_uint16 format = ttUSHORT(data + index_map + 0);
797 if (format == 0) { // apple byte encoding
798 stbtt_int32 bytes = ttUSHORT(data + index_map + 2);
799 if (unicode_codepoint < bytes-6)
800 return ttBYTE(data + index_map + 6 + unicode_codepoint);
801 return 0;
802 } else if (format == 6) {
803 stbtt_uint32 first = ttUSHORT(data + index_map + 6);
804 stbtt_uint32 count = ttUSHORT(data + index_map + 8);
805 if ((stbtt_uint32) unicode_codepoint >= first && (stbtt_uint32) unicode_codepoint < first+count)
806 return ttUSHORT(data + index_map + 10 + (unicode_codepoint - first)*2);
807 return 0;
808 } else if (format == 2) {
809 STBTT_assert(0); // @TODO: high-byte mapping for japanese/chinese/korean
810 return 0;
811 } else if (format == 4) { // standard mapping for windows fonts: binary search collection of ranges
812 stbtt_uint16 segcount = ttUSHORT(data+index_map+6) >> 1;
813 stbtt_uint16 searchRange = ttUSHORT(data+index_map+8) >> 1;
814 stbtt_uint16 entrySelector = ttUSHORT(data+index_map+10);
815 stbtt_uint16 rangeShift = ttUSHORT(data+index_map+12) >> 1;
816 stbtt_uint16 item, offset, start, end;
817
818 // do a binary search of the segments
819 stbtt_uint32 endCount = index_map + 14;
820 stbtt_uint32 search = endCount;
821
822 if (unicode_codepoint > 0xffff)
823 return 0;
824
825 // they lie from endCount .. endCount + segCount
826 // but searchRange is the nearest power of two, so...
827 if (unicode_codepoint >= ttUSHORT(data + search + rangeShift*2))
828 search += rangeShift*2;
829
830 // now decrement to bias correctly to find smallest
831 search -= 2;
832 while (entrySelector) {
833 stbtt_uint16 start, end;
834 searchRange >>= 1;
835 start = ttUSHORT(data + search + 2 + segcount*2 + 2);
836 end = ttUSHORT(data + search + 2);
837 start = ttUSHORT(data + search + searchRange*2 + segcount*2 + 2);
838 end = ttUSHORT(data + search + searchRange*2);
839 if (unicode_codepoint > end)
840 search += searchRange*2;
841 --entrySelector;
842 }
843 search += 2;
844
845 item = (stbtt_uint16) ((search - endCount) >> 1);
846
847 STBTT_assert(unicode_codepoint <= ttUSHORT(data + endCount + 2*item));
848 start = ttUSHORT(data + index_map + 14 + segcount*2 + 2 + 2*item);
849 end = ttUSHORT(data + index_map + 14 + 2 + 2*item);
850 if (unicode_codepoint < start)
851 return 0;
852
853 offset = ttUSHORT(data + index_map + 14 + segcount*6 + 2 + 2*item);
854 if (offset == 0)
855 return (stbtt_uint16) (unicode_codepoint + ttSHORT(data + index_map + 14 + segcount*4 + 2 + 2*item));
856
857 return ttUSHORT(data + offset + (unicode_codepoint-start)*2 + index_map + 14 + segcount*6 + 2 + 2*item);
858 } else if (format == 12 || format == 13) {
859 stbtt_uint32 ngroups = ttULONG(data+index_map+12);
860 stbtt_int32 low,high;
861 stbtt_uint16 g = 0;
862 low = 0; high = (stbtt_int32)ngroups;
863 // Binary search the right group.
864 while (low < high) {
865 stbtt_int32 mid = low + ((high-low) >> 1); // rounds down, so low <= mid < high
866 stbtt_uint32 start_char = ttULONG(data+index_map+16+mid*12);
867 stbtt_uint32 end_char = ttULONG(data+index_map+16+mid*12+4);
868 if ((stbtt_uint32) unicode_codepoint < start_char)
869 high = mid;
870 else if ((stbtt_uint32) unicode_codepoint > end_char)
871 low = mid+1;
872 else {
873 stbtt_uint32 start_glyph = ttULONG(data+index_map+16+mid*12+8);
874 if (format == 12)
875 return start_glyph + unicode_codepoint-start_char;
876 else // format == 13
877 return start_glyph;
878 }
879 }
880 return 0; // not found
881 }
882 // @TODO
883 STBTT_assert(0);
884 return 0;
885}

952int stbtt_GetGlyphShape(const stbtt_fontinfo *info, int glyph_index, stbtt_vertex **pvertices)
953{
954 stbtt_int16 numberOfContours;
955 stbtt_uint8 *endPtsOfContours;
956 stbtt_uint8 *data = info->data;
957 stbtt_vertex *vertices=0;
958 int num_vertices=0;
959 int g = stbtt__GetGlyfOffset(info, glyph_index);
960
961 *pvertices = NULL;
962
963 if (g < 0) return 0;
964
965 numberOfContours = ttSHORT(data + g);
966
967 if (numberOfContours > 0) {
968 stbtt_uint8 flags=0,flagcount;
969 stbtt_int32 ins, i,j=0,m,n, next_move, was_off=0, off, start_off=0, curve_end=0;
970 stbtt_int32 x,y,cx,cy,sx,sy, scx,scy;
971 stbtt_uint8 *points;
972 endPtsOfContours = (data + g + 10);
973 ins = ttUSHORT(data + g + 10 + numberOfContours * 2);
974 points = data + g + 10 + numberOfContours * 2 + 2 + ins;
975
976 n = 1+ttUSHORT(endPtsOfContours + numberOfContours*2-2);
977
978 m = n + 2*numberOfContours; // a loose bound on how many vertices we might need
979 vertices = (stbtt_vertex *) STBTT_malloc(m * sizeof(vertices[0]), info->userdata);
980 if (vertices == 0)
981 return 0;
982
983 next_move = 0;
984 flagcount=0;
985
986 // in first pass, we load uninterpreted data into the allocated array
987 // above, shifted to the end of the array so we won't overwrite it when
988 // we create our final data starting from the front
989
990 off = m - n; // starting offset for uninterpreted data, regardless of how m ends up being calculated
991
992 // first load flags
993
994 for (i=0; i < n; ++i) {
995 if (flagcount == 0) {
996 flags = *points++;
997 if (flags & 8)
998 flagcount = *points++;
999 } else
1000 --flagcount;
1001 vertices[off+i].type = flags;
1002 }
1003
1004 // now load x coordinates
1005 x=0;
1006 for (i=0; i < n; ++i) {
1007 flags = vertices[off+i].type;
1008 if (flags & 2) {
1009 stbtt_int16 dx = *points++;
1010 x += (flags & 16) ? dx : -dx; // ???
1011 } else {
1012 if (!(flags & 16)) {
1013 x = x + (stbtt_int16) (points[0]*256 + points[1]);
1014 points += 2;
1015 }
1016 }
1017 vertices[off+i].x = x;
1018 }
1019
1020 // now load y coordinates
1021 y=0;
1022 for (i=0; i < n; ++i) {
1023 flags = vertices[off+i].type;
1024 if (flags & 4) {
1025 stbtt_int16 dy = *points++;
1026 y += (flags & 32) ? dy : -dy; // ???
1027 } else {
1028 if (!(flags & 32)) {
1029 y = y + (stbtt_int16) (points[0]*256 + points[1]);
1030 points += 2;
1031 }
1032 }
1033 vertices[off+i].y = y;
1034 }
1035
1036 // now convert them to our format
1037 num_vertices=0;
1038 sx = sy = cx = cy = 0;
1039 for (i=0; i < n; ++i) {
1040 flags = vertices[off+i].type;
1041 x = (stbtt_int16) vertices[off+i].x;
1042 y = (stbtt_int16) vertices[off+i].y;
1043
1044 if (next_move == i) {
1045 if (i != 0)
1046 num_vertices = stbtt__close_shape(vertices, num_vertices, was_off, start_off, sx,sy,scx,scy,cx,cy);
1047
1048 // now start the new one
1049 start_off = !(flags & 1);
1050 if (start_off) {
1051 // if we start off with an off-curve point, then when we need to find a point on the curve
1052 // where we can start, and we need to save some state for when we wraparound.
1053 scx = x;
1054 scy = y;
1055 if (!(vertices[off+i+1].type & 1)) {
1056 // next point is also a curve point, so interpolate an on-point curve
1057 sx = (x + (stbtt_int32) vertices[off+i+1].x) >> 1;
1058 sy = (y + (stbtt_int32) vertices[off+i+1].y) >> 1;
1059 } else {
1060 // otherwise just use the next point as our start point
1061 sx = (stbtt_int32) vertices[off+i+1].x;
1062 sy = (stbtt_int32) vertices[off+i+1].y;
1063 ++i; // we're using point i+1 as the starting point, so skip it
1064 }
1065 } else {
1066 sx = x;
1067 sy = y;
1068 }
1069 stbtt_setvertex(&vertices[num_vertices++], STBTT_vmove,sx,sy,0,0);
1070 was_off = 0;
1071 next_move = 1 + ttUSHORT(endPtsOfContours+j*2);
1072 ++j;
1073 } else {
1074 if (!(flags & 1)) { // if it's a curve
1075 if (was_off) // two off-curve control points in a row means interpolate an on-curve midpoint
1076 stbtt_setvertex(&vertices[num_vertices++], STBTT_vcurve, (cx+x)>>1, (cy+y)>>1, cx, cy);
1077 cx = x;
1078 cy = y;
1079 was_off = 1;
1080 } else {
1081 if (was_off)
1082 stbtt_setvertex(&vertices[num_vertices++], STBTT_vcurve, x,y, cx, cy);
1083 else
1084 stbtt_setvertex(&vertices[num_vertices++], STBTT_vline, x,y,0,0);
1085 was_off = 0;
1086 }
1087 }
1088 }
1089 num_vertices = stbtt__close_shape(vertices, num_vertices, was_off, start_off, sx,sy,scx,scy,cx,cy);
1090 } else if (numberOfContours == -1) {
1091 // Compound shapes.
1092 int more = 1;
1093 stbtt_uint8 *comp = data + g + 10;
1094 num_vertices = 0;
1095 vertices = 0;
1096 while (more) {
1097 stbtt_uint16 flags, gidx;
1098 int comp_num_verts = 0, i;
1099 stbtt_vertex *comp_verts = 0, *tmp = 0;
1100 float mtx[6] = {1,0,0,1,0,0}, m, n;
1101
1102 flags = ttSHORT(comp); comp+=2;
1103 gidx = ttSHORT(comp); comp+=2;
1104
1105 if (flags & 2) { // XY values
1106 if (flags & 1) { // shorts
1107 mtx[4] = ttSHORT(comp); comp+=2;
1108 mtx[5] = ttSHORT(comp); comp+=2;
1109 } else {
1110 mtx[4] = ttCHAR(comp); comp+=1;
1111 mtx[5] = ttCHAR(comp); comp+=1;
1112 }
1113 }
1114 else {
1115 // @TODO handle matching point
1116 STBTT_assert(0);
1117 }
1118 if (flags & (1<<3)) { // WE_HAVE_A_SCALE
1119 mtx[0] = mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;
1120 mtx[1] = mtx[2] = 0;
1121 } else if (flags & (1<<6)) { // WE_HAVE_AN_X_AND_YSCALE
1122 mtx[0] = ttSHORT(comp)/16384.0f; comp+=2;
1123 mtx[1] = mtx[2] = 0;
1124 mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;
1125 } else if (flags & (1<<7)) { // WE_HAVE_A_TWO_BY_TWO
1126 mtx[0] = ttSHORT(comp)/16384.0f; comp+=2;
1127 mtx[1] = ttSHORT(comp)/16384.0f; comp+=2;
1128 mtx[2] = ttSHORT(comp)/16384.0f; comp+=2;
1129 mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;
1130 }
1131
1132 // Find transformation scales.
1133 m = (float) sqrt(mtx[0]*mtx[0] + mtx[1]*mtx[1]);
1134 n = (float) sqrt(mtx[2]*mtx[2] + mtx[3]*mtx[3]);
1135
1136 // Get indexed glyph.
1137 comp_num_verts = stbtt_GetGlyphShape(info, gidx, &comp_verts);
1138 if (comp_num_verts > 0) {
1139 // Transform vertices.
1140 for (i = 0; i < comp_num_verts; ++i) {
1141 stbtt_vertex* v = &comp_verts[i];
1142 stbtt_vertex_type x,y;
1143 x=v->x; y=v->y;
1144 v->x = (stbtt_vertex_type)(m * (mtx[0]*x + mtx[2]*y + mtx[4]));
1145 v->y = (stbtt_vertex_type)(n * (mtx[1]*x + mtx[3]*y + mtx[5]));
1146 x=v->cx; y=v->cy;
1147 v->cx = (stbtt_vertex_type)(m * (mtx[0]*x + mtx[2]*y + mtx[4]));
1148 v->cy = (stbtt_vertex_type)(n * (mtx[1]*x + mtx[3]*y + mtx[5]));
1149 }
1150 // Append vertices.
1151 tmp = (stbtt_vertex*)STBTT_malloc((num_vertices+comp_num_verts)*sizeof(stbtt_vertex), info->userdata);
1152 if (!tmp) {
1153 if (vertices) STBTT_free(vertices, info->userdata);
1154 if (comp_verts) STBTT_free(comp_verts, info->userdata);
1155 return 0;
1156 }
1157 if (num_vertices > 0) memcpy(tmp, vertices, num_vertices*sizeof(stbtt_vertex));
1158 memcpy(tmp+num_vertices, comp_verts, comp_num_verts*sizeof(stbtt_vertex));
1159 if (vertices) STBTT_free(vertices, info->userdata);
1160 vertices = tmp;
1161 STBTT_free(comp_verts, info->userdata);
1162 num_vertices += comp_num_verts;
1163 }
1164 // More components ?
1165 more = flags & (1<<5);
1166 }
1167 } else if (numberOfContours < 0) {
1168 // @TODO other compound variations?
1169 STBTT_assert(0);
1170 } else {
1171 // numberOfCounters == 0, do nothing
1172 }
1173
1174 *pvertices = vertices;
1175 return num_vertices;
1176}

1286typedef struct stbtt__active_edge
1287{
1288 int x,dx;
1289 float ey;
1290 struct stbtt__active_edge *next;
1291 int valid;

1294#define FIXSHIFT 10
1295#define FIX (1 << FIXSHIFT)
1296#define FIXMASK (FIX-1)
1297
1298static stbtt__active_edge *new_active(stbtt__edge *e, int off_x, float start_point, void *userdata)
1299{
1300 stbtt__active_edge *z = (stbtt__active_edge *) STBTT_malloc(sizeof(*z), userdata); // @TODO: make a pool of these!!!
1301 float dxdy = (e->x1 - e->x0) / (e->y1 - e->y0);
1302 STBTT_assert(e->y0 <= start_point);
1303 if (!z) return z;
1304 // round dx down to avoid going too far
1305 if (dxdy < 0)
1306 z->dx = -STBTT_ifloor(FIX * -dxdy);
1307 else
1308 z->dx = STBTT_ifloor(FIX * dxdy);
1309 z->x = STBTT_ifloor(FIX * (e->x0 + dxdy * (start_point - e->y0)));
1310 z->x -= off_x * FIX;
1311 z->ey = e->y1;
1312 z->next = 0;
1313 z->valid = e->invert ? 1 : -1;
1314 return z;
1315}

1317// note: this routine clips fills that extend off the edges... ideally this
1318// wouldn't happen, but it could happen if the truetype glyph bounding boxes
1319// are wrong, or if the user supplies a too-small bitmap
1320static void stbtt__fill_active_edges(unsigned char *scanline, int len, stbtt__active_edge *e, int max_weight)
1321{
1322 // non-zero winding fill
1323 int x0=0, w=0;
1324
1325 while (e) {
1326 if (w == 0) {
1327 // if we're currently at zero, we need to record the edge start point
1328 x0 = e->x; w += e->valid;
1329 } else {
1330 int x1 = e->x; w += e->valid;
1331 // if we went to zero, we need to draw
1332 if (w == 0) {
1333 int i = x0 >> FIXSHIFT;
1334 int j = x1 >> FIXSHIFT;
1335
1336 if (i < len && j >= 0) {
1337 if (i == j) {
1338 // x0,x1 are the same pixel, so compute combined coverage
1339 scanline[i] = scanline[i] + (stbtt_uint8) ((x1 - x0) * max_weight >> FIXSHIFT);
1340 } else {
1341 if (i >= 0) // add antialiasing for x0
1342 scanline[i] = scanline[i] + (stbtt_uint8) (((FIX - (x0 & FIXMASK)) * max_weight) >> FIXSHIFT);
1343 else
1344 i = -1; // clip
1345
1346 if (j < len) // add antialiasing for x1
1347 scanline[j] = scanline[j] + (stbtt_uint8) (((x1 & FIXMASK) * max_weight) >> FIXSHIFT);
1348 else
1349 j = len; // clip
1350
1351 for (++i; i < j; ++i) // fill pixels between x0 and x1
1352 scanline[i] = scanline[i] + (stbtt_uint8) max_weight;
1353 }
1354 }
1355 }
1356 }
1357
1358 e = e->next;
1359 }
1360}

1362static void stbtt__rasterize_sorted_edges(stbtt__bitmap *result, stbtt__edge *e, int n, int vsubsample, int off_x, int off_y, void *userdata)
1363{
1364 stbtt__active_edge *active = NULL;
1365 int y,j=0;
1366 int max_weight = (255 / vsubsample); // weight per vertical scanline
1367 int s; // vertical subsample index
1368 unsigned char scanline_data[512], *scanline;
1369
1370 if (result->w > 512)
1371 scanline = (unsigned char *) STBTT_malloc(result->w, userdata);
1372 else
1373 scanline = scanline_data;
1374
1375 y = off_y * vsubsample;
1376 e[n].y0 = (off_y + result->h) * (float) vsubsample + 1;
1377
1378 while (j < result->h) {
1379 STBTT_memset(scanline, 0, result->w);
1380 for (s=0; s < vsubsample; ++s) {
1381 // find center of pixel for this scanline
1382 float scan_y = y + 0.5f;
1383 stbtt__active_edge **step = &active;
1384
1385 // update all active edges;
1386 // remove all active edges that terminate before the center of this scanline
1387 while (*step) {
1388 stbtt__active_edge * z = *step;
1389 if (z->ey <= scan_y) {
1390 *step = z->next; // delete from list
1391 STBTT_assert(z->valid);
1392 z->valid = 0;
1393 STBTT_free(z, userdata);
1394 } else {
1395 z->x += z->dx; // advance to position for current scanline
1396 step = &((*step)->next); // advance through list
1397 }
1398 }
1399
1400 // resort the list if needed
1401 for(;;) {
1402 int changed=0;
1403 step = &active;
1404 while (*step && (*step)->next) {
1405 if ((*step)->x > (*step)->next->x) {
1406 stbtt__active_edge *t = *step;
1407 stbtt__active_edge *q = t->next;
1408
1409 t->next = q->next;
1410 q->next = t;
1411 *step = q;
1412 changed = 1;
1413 }
1414 step = &(*step)->next;
1415 }
1416 if (!changed) break;
1417 }
1418
1419 // insert all edges that start before the center of this scanline -- omit ones that also end on this scanline
1420 while (e->y0 <= scan_y) {
1421 if (e->y1 > scan_y) {
1422 stbtt__active_edge *z = new_active(e, off_x, scan_y, userdata);
1423 // find insertion point
1424 if (active == NULL)
1425 active = z;
1426 else if (z->x < active->x) {
1427 // insert at front
1428 z->next = active;
1429 active = z;
1430 } else {
1431 // find thing to insert AFTER
1432 stbtt__active_edge *p = active;
1433 while (p->next && p->next->x < z->x)
1434 p = p->next;
1435 // at this point, p->next->x is NOT < z->x
1436 z->next = p->next;
1437 p->next = z;
1438 }
1439 }
1440 ++e;
1441 }
1442
1443 // now process all active edges in XOR fashion
1444 if (active)
1445 stbtt__fill_active_edges(scanline, result->w, active, max_weight);
1446
1447 ++y;
1448 }
1449 STBTT_memcpy(result->pixels + j * result->stride, scanline, result->w);
1450 ++j;
1451 }
1452
1453 while (active) {
1454 stbtt__active_edge *z = active;
1455 active = active->next;
1456 STBTT_free(z, userdata);
1457 }
1458
1459 if (scanline != scanline_data)
1460 STBTT_free(scanline, userdata);
1461}

1478static void stbtt__rasterize(stbtt__bitmap *result, stbtt__point *pts, int *wcount, int windings, float scale_x, float scale_y, float shift_x, float shift_y, int off_x, int off_y, int invert, void *userdata)
1479{
1480 float y_scale_inv = invert ? -scale_y : scale_y;
1481 stbtt__edge *e;
1482 int n,i,j,k,m;
1483 int vsubsample = result->h < 8 ? 15 : 5;
1484 // vsubsample should divide 255 evenly; otherwise we won't reach full opacity
1485
1486 // now we have to blow out the windings into explicit edge lists
1487 n = 0;
1488 for (i=0; i < windings; ++i)
1489 n += wcount[i];
1490
1491 e = (stbtt__edge *) STBTT_malloc(sizeof(*e) * (n+1), userdata); // add an extra one as a sentinel
1492 if (e == 0) return;
1493 n = 0;
1494
1495 m=0;
1496 for (i=0; i < windings; ++i) {
1497 stbtt__point *p = pts + m;
1498 m += wcount[i];
1499 j = wcount[i]-1;
1500 for (k=0; k < wcount[i]; j=k++) {
1501 int a=k,b=j;
1502 // skip the edge if horizontal
1503 if (p[j].y == p[k].y)
1504 continue;
1505 // add edge from j to k to the list
1506 e[n].invert = 0;
1507 if (invert ? p[j].y > p[k].y : p[j].y < p[k].y) {
1508 e[n].invert = 1;
1509 a=j,b=k;
1510 }
1511 e[n].x0 = p[a].x * scale_x + shift_x;
1512 e[n].y0 = p[a].y * y_scale_inv * vsubsample + shift_y;
1513 e[n].x1 = p[b].x * scale_x + shift_x;
1514 e[n].y1 = p[b].y * y_scale_inv * vsubsample + shift_y;
1515 ++n;
1516 }
1517 }
1518
1519 // now sort the edges by their highest point (should snap to integer, and then by x)
1520 STBTT_sort(e, n, sizeof(e[0]), stbtt__edge_compare);
1521
1522 // now, traverse the scanlines and find the intersections on each scanline, use xor winding rule
1523 stbtt__rasterize_sorted_edges(result, e, n, vsubsample, off_x, off_y, userdata);
1524
1525 STBTT_free(e, userdata);
1526}

1556// returns number of contours
1557stbtt__point *stbtt_FlattenCurves(stbtt_vertex *vertices, int num_verts, float objspace_flatness, int **contour_lengths, int *num_contours, void *userdata)
1558{
1559 stbtt__point *points=0;
1560 int num_points=0;
1561
1562 float objspace_flatness_squared = objspace_flatness * objspace_flatness;
1563 int i,n=0,start=0, pass;
1564
1565 // count how many "moves" there are to get the contour count
1566 for (i=0; i < num_verts; ++i)
1567 if (vertices[i].type == STBTT_vmove)
1568 ++n;
1569
1570 *num_contours = n;
1571 if (n == 0) return 0;
1572
1573 *contour_lengths = (int *) STBTT_malloc(sizeof(**contour_lengths) * n, userdata);
1574
1575 if (*contour_lengths == 0) {
1576 *num_contours = 0;
1577 return 0;
1578 }
1579
1580 // make two passes through the points so we don't need to realloc
1581 for (pass=0; pass < 2; ++pass) {
1582 float x=0,y=0;
1583 if (pass == 1) {
1584 points = (stbtt__point *) STBTT_malloc(num_points * sizeof(points[0]), userdata);
1585 if (points == NULL) goto error;
1586 }
1587 num_points = 0;
1588 n= -1;
1589 for (i=0; i < num_verts; ++i) {
1590 switch (vertices[i].type) {
1591 case STBTT_vmove:
1592 // start the next contour
1593 if (n >= 0)
1594 (*contour_lengths)[n] = num_points - start;
1595 ++n;
1596 start = num_points;
1597
1598 x = vertices[i].x, y = vertices[i].y;
1599 stbtt__add_point(points, num_points++, x,y);
1600 break;
1601 case STBTT_vline:
1602 x = vertices[i].x, y = vertices[i].y;
1603 stbtt__add_point(points, num_points++, x, y);
1604 break;
1605 case STBTT_vcurve:
1606 stbtt__tesselate_curve(points, &num_points, x,y,
1607 vertices[i].cx, vertices[i].cy,
1608 vertices[i].x, vertices[i].y,
1609 objspace_flatness_squared, 0);
1610 x = vertices[i].x, y = vertices[i].y;
1611 break;
1612 }
1613 }
1614 (*contour_lengths)[n] = num_points - start;
1615 }
1616
1617 return points;
1618error:
1619 STBTT_free(points, userdata);
1620 STBTT_free(*contour_lengths, userdata);
1621 *contour_lengths = 0;
1622 *num_contours = 0;
1623 return NULL;

1805// check if a utf8 string contains a prefix which is the utf16 string; if so return length of matching utf8 string
1806static stbtt_int32 stbtt__CompareUTF8toUTF16_bigendian_prefix(const stbtt_uint8 *s1, stbtt_int32 len1, const stbtt_uint8 *s2, stbtt_int32 len2)
1807{
1808 stbtt_int32 i=0;
1809
1810 // convert utf16 to utf8 and compare the results while converting
1811 while (len2) {
1812 stbtt_uint16 ch = s2[0]*256 + s2[1];
1813 if (ch < 0x80) {
1814 if (i >= len1) return -1;
1815 if (s1[i++] != ch) return -1;
1816 } else if (ch < 0x800) {
1817 if (i+1 >= len1) return -1;
1818 if (s1[i++] != 0xc0 + (ch >> 6)) return -1;
1819 if (s1[i++] != 0x80 + (ch & 0x3f)) return -1;
1820 } else if (ch >= 0xd800 && ch < 0xdc00) {
1821 stbtt_uint32 c;
1822 stbtt_uint16 ch2 = s2[2]*256 + s2[3];
1823 if (i+3 >= len1) return -1;
1824 c = ((ch - 0xd800) << 10) + (ch2 - 0xdc00) + 0x10000;
1825 if (s1[i++] != 0xf0 + (c >> 18)) return -1;
1826 if (s1[i++] != 0x80 + ((c >> 12) & 0x3f)) return -1;
1827 if (s1[i++] != 0x80 + ((c >> 6) & 0x3f)) return -1;
1828 if (s1[i++] != 0x80 + ((c ) & 0x3f)) return -1;
1829 s2 += 2; // plus another 2 below
1830 len2 -= 2;
1831 } else if (ch >= 0xdc00 && ch < 0xe000) {
1832 return -1;
1833 } else {
1834 if (i+2 >= len1) return -1;
1835 if (s1[i++] != 0xe0 + (ch >> 12)) return -1;
1836 if (s1[i++] != 0x80 + ((ch >> 6) & 0x3f)) return -1;
1837 if (s1[i++] != 0x80 + ((ch ) & 0x3f)) return -1;
1838 }
1839 s2 += 2;
1840 len2 -= 2;
1841 }
1842 return i;
1843}

1883 // find the encoding

1886 // is this a Unicode encoding?

1904 } else {
1905 // if nothing immediately following
1906 if (matchlen == nlen)
1907 return 1;
1908 }
1909 }
1910 }

1913 }
1914 }

1918static int stbtt__matches(stbtt_uint8 *fc, stbtt_uint32 offset, stbtt_uint8 *name, stbtt_int32 flags)
1919{
1920 stbtt_int32 nlen = STBTT_strlen((char *) name);
1921 stbtt_uint32 nm,hd;
1922 if (!stbtt__isfont(fc+offset)) return 0;
1923
1924 // check italics/bold/underline flags in macStyle...
1925 if (flags) {
1926 hd = stbtt__find_table(fc, offset, "head");

1929
1930 nm = stbtt__find_table(fc, offset, "name");
1931 if (!nm) return 0;
1932

1935 if (stbtt__matchpair(fc, nm, name, nlen, 16, -1)) return 1;
1936 if (stbtt__matchpair(fc, nm, name, nlen, 1, -1)) return 1;
1937 if (stbtt__matchpair(fc, nm, name, nlen, 3, -1)) return 1;

1940 if (stbtt__matchpair(fc, nm, name, nlen, 1, 2)) return 1;
1941 if (stbtt__matchpair(fc, nm, name, nlen, 3, -1)) return 1;
1942 }
1943
1944 return 0;
